**app.js** (JavaScript)
```javascript
// Constantes para los elementos HTML
const btnSumar = document.getElementById('btnSumar');
const btnRestar = document.getElementById('btnRestar');
const btnMultiplicar = document.getElementById('btnMultiplicar');
const btnDividir = document.getElementById('btnDividir');
const resultado = document.getElementById('resultado');
const tecladoNumerico = document.getElementById('tecladoNumerico');
const tecladoConfiguracion = document.getElementById('tecladoConfiguracion');

// Objetos para los numeros y el operador
let numero1, numero2;
let operador;

// Función para mostrar o esconder el teclado numérico
function mostrarTeclado() {
  if (tecladoNumerico.style.display === 'none') {
    tecladoNumerico.style.display = 'block';
    tecladoConfiguracion.style.display = 'none';
  } else {
    tecladoNumerico.reset();
    tecladoNumerico.style.display = 'none';
  }
}

// Función para realizar la operación de suma
function sumar() {
  mostrarTeclado();
  operador = '+';
}

// Función para realizar la operación de resta
function restar() {
  mostrarTeclado();
  operador = '-';
}

// Función para realizar la operación de multiplicación
function multiplicar() {
  mostrarTeclado();
  operador = '*';
}

// Función para realizar la operación de división
function dividir() {
  mostrarTeclado();
  operador = '/';
}
```
