3. app.js
```javascript
const btnSumar = document.getElementById('btnSumar');
const btnRestar = document.getElementById('btnRestar');
const btnMultiplicar = document.getElementById('btnMultiplicar');
const btnDividir = document.getElementById('btnDividir');
const resultado = document.getElementById('resultado');
const tecladoNumerico = document.getElementById('tecladoNumerico');
const operador = document.getElementById('operador');

let numero1, numero2;

// Agrega un evento click a los botones de operación que muestre el teclado numérico con dos campos para ingresar los números
function mostrarTecladoNumérico() {
  tecladoNumerico.style.display = "block";
}

btnSumar.addEventListener("click", () => {
  mostrarTecladoNumérico();
  numero1 = resultado.innerText;
  operador.value = '+';
});

btnRestar.addEventListener("click", () => {
  mostrarTecladoNumérico();
  numero1 = resultado.innerText;
  operador.value = '-';
});

btnMultiplicar.addEventListener("click", () => {
  mostrarTecladoNumérico();
  numero1 = resultado.innerText;
  operador.value = '*';
});

btnDividir.addEventListener("click", () => {
  mostrarTecladoNumérico();
  numero1 = resultado.innerText;
  operador.value = '/';
});

// Función para procesar las operaciones matemáticas y mostrar el resultado en pantalla
function calcular() {
  const numero2Input = document.getElementById('numero2');
  numero2 = numero2Input.value;
  fetch('/api/calculadora', {
    method: 'POST',
    headers: {
      "Content-Type": "application/json",
    },
    body: JSON.stringify({ operador, numero1, numero2 }),
  })
    .then((response) => response.json())
    .then((resultadoJson) => {
      resultado.innerText = resultadoJson.resultado;
      tecladoNumerico.style.display = "none";
    });
}
```
